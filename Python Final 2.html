def drivingDist (series):
    if series == "Not Distracted":
        return "Not"
    if series == "Looked But Did Not See":
        return "Inattentive"
    if series == "By Other Occupant(s)":
        return "Occupant"
    if series == "By a Moving Object in Vehicle":
        return "Interior Vehicle Activity"
    if series == "While Talking or Listening to Cellular Phone":
        return "Cell Phone"
    if series == "While Manipulating Cellular Phone":
        return "Cell Phone"
    if series == "While Adjusting Audio or Climate Controls":
        return "Interior Vehicle Activity"
    if series == "While Using Other Component/Controls Integral to Vehicle":
        return "Interior Vehicle Activity"
    if series == "While Using or Reaching For Device/Object Brought Into Vehicle":
        return "Interior Vehicle Activity"
    if series == "Distracted by Outside Person, Object or Event":
        return "Other"
    if series == "Eating or Drinking":
        return "Eat and Drink"
    if series == "Smoking Related":
        return "Smoking"
    if series == "Other Cellular Phone Related":
        return "Cell Phone"
    if series == "No Driver Present/Unknown if Driver Present":
        return "Not"
    if series == "Distraction/Inattention":
        return "Inattentive"
    if series == "Distraction/Careless":
        return "Inattentive"
    if series == "Careless/Inattentive":
        return "Inattentive"
    if series == "Distraction (Distracted), Details Unknown":
        return "Inattentive"
    if series == "Inattention (Inattentive), Details Unknown":
        return "Inattentive"
    if series == "Not Reported":
        return "Unknown"
    if series == "Lost In Thought/Day Dreaming":
        return "Other"
    if series == "Other Distraction":
        return "Other"
    if series == "Unknown if Distracted":
        return "Unknown"
    distract1['drivingDist']=distract1['DISTRACTION'].apply(drivingDist)
